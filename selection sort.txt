#include <stdio.h>

int main()
{
    int n,i,j,arr[100],temp;
    printf("\n enter no of elements:");
    scanf("%d",&n);
    printf("\nenter %d integers",n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
    }
    for(i=0;i<n;i++)
    {
        for(j=i+1;j<n;j++)
        {
            if(arr[i]>arr[j])
            {
                temp=arr[i];
                arr[i]=arr[j];
                arr[j]=temp;
            }
        }
    }
    printf("\nthe sorted order is");
    for(int k=0;k<n;k++)
    {
        printf("%d ",arr[k]);
    }
}
